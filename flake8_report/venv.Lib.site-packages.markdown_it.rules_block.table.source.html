21 def escapedSplit(string: str) -> list[str]:
22 result: list[str] = []
23 pos = 0
24 max = len(string)
25 isEscaped = False
26 lastPos = 0
27 current = ""
28 ch = charStrAt(string, pos)
29  
30 while pos < max:
31 if ch == "|":
32 if not isEscaped:
33 # pipe separating cells, '|'
34 result.append(current + string[lastPos:pos])
35 current = ""
36 lastPos = pos + 1
37 else:
38 # escaped pipe, '\|'
  • E203 Whitespace before ':'
39 current += string[lastPos : pos - 1]
40 lastPos = pos
41  
42 isEscaped = ch == "\\"
43 pos += 1
44  
45 ch = charStrAt(string, pos)
